Assistentenschritte

Perplexity und ChatGPT unterscheiden sich vor allem in ihrer Funktionsweise und ihrem Einsatzfokus:

  • Perplexity AI ist darauf spezialisiert, aktuelle und faktenbasierte Antworten zu liefern, indem es in Echtzeit auf vielfltige Quellen im Internet zugreift und die Herkunft der Informationen transparent mit Quellenangaben belegt. Es bietet eine suchmaschinenhnliche Funktionalitt und eignet sich besonders fr Forschungs- und Recherchezwecke bei Fachleuten und Studierenden, die verifizierbare Informationen bentigen.

  • ChatGPT basiert auf einem groen neuronalen Sprachmodell, das auf umfangreichen Trainingsdaten bis zu einem gewissen Zeitpunkt trainiert wurde. Es erzeugt menschenhnliche, kontextbezogene Dialoge mit hoher Vielseitigkeit, von kreativen Texten bis zu komplexen Dialogen. ChatGPT ist besonders stark in der Multimodalitt (Text, Bild, Audio), kreativen Anwendungen und interaktiven Nutzererlebnissen, verzichtet aber auf direkte Quellenangaben und liefert nicht immer aktuelle Echtzeitinformationen.

Zusammengefasst: Perplexity punktet mit przisen, zitierfhigen Echtzeitdaten fr faktische Recherche, whrend ChatGPT mit kreativer Textgenerierung, komplexem kontextuellem Verstndnis und interaktiven Gesprchserfahrungen berzeugt. <script>
  // Laden der Tasks aus localStorage oder leere Listen
  let tasks = JSON.parse(localStorage.getItem('kanbanTasks')) || {
    todo: [],
    inprogress: [],
    done: []
  };

  function saveTasks() {
    localStorage.setItem('kanbanTasks', JSON.stringify(tasks));
  }

  // Aufgabe hinzufgen
  function addTask(status) {
    const taskText = prompt('Neue Aufgabe eingeben:');
    if (taskText && taskText.trim() !== '') {
      tasks[status].push(taskText.trim());
      saveTasks();
      renderTasks();
    }
  }

  // Aufgabe lschen
  function deleteTask(status, index) {
    if (confirm('Aufgabe lschen?')) {
      tasks[status].splice(index, 1);
      saveTasks();
      renderTasks();
    }
  }

  // Aufgabe bearbeiten
  function editTask(status, index, event) {
    tasks[status][index] = event.target.value;
    saveTasks();
  }

  // Drag & Drop Funktionen
  let dragged = null;

  function dragStart(event, status, index) {
    dragged = {status, index};
    event.dataTransfer.effectAllowed = 'move';
  }

  function dragOver(event) {
    event.preventDefault();
  }

  function drop(event, newStatus) {
    event.preventDefault();
    if (dragged) {
      const taskText = tasks[dragged.status][dragged.index];
      tasks[dragged.status].splice(dragged.index, 1);
      tasks[newStatus].push(taskText);
      saveTasks();
      renderTasks();
      dragged = null;
    }
  }

  // Aufgaben rendern
  function renderTasks() {
    ['todo', 'inprogress', 'done'].forEach(status => {
      const container = document.getElementById(status + 'Tasks');
      container.innerHTML = '';
      tasks[status].forEach((taskText, index) => {
        const task = document.createElement('div');
        task.className = 'task';
        task.setAttribute('draggable', 'true');
        task.ondragstart = (e) => dragStart(e, status, index);

        const textarea = document.createElement('textarea');
        textarea.value = taskText;
        textarea.rows = 2;
        textarea.oninput = (e) => editTask(status, index, e);
        textarea.style.border = "none";
        textarea.style.background = "transparent";
        textarea.style.width = "100%";
        textarea.style.resize = "none";
        
        const delBtn = document.createElement('button');
        delBtn.textContent = 'X';
        delBtn.title = 'Lschen';
        delBtn.style.float = 'right';
        delBtn.style.marginLeft = '5px';
        delBtn.onclick = () => deleteTask(status, index);

        task.appendChild(delBtn);
        task.appendChild(textarea);
        container.appendChild(task);
      });
    });
  }

  // Drag & Drop Event Listeners fr Spalten
  document.querySelectorAll('.column').forEach(col => {
    col.ondragover = dragOver;
    col.ondrop = e => drop(e, col.getAttribute('data-status'));
  });

  renderTasks();
</script>
</body>

Fr das Hosting eures einfachen Kanban-Boards gibt es einige unkomplizierte und kostengnstige Mglichkeiten:

  • GitHub Pages: Ideal, wenn ihr keinen eigenen Server habt. Ihr ladet die kanban.html-Datei in ein kostenloses GitHub-Repository hoch und aktiviert GitHub Pages. So ist die App online und ihr knnt gemeinsam darauf zugreifen.

  • Netlify / Vercel: Beide bieten kostenlose Plne fr statische Websites. Ihr ladet die Datei einfach hoch, und schon ist die App online verfgbar mit einer eigenen URL.

  • Webspace beim Anbieter: Falls ihr einen eigenen Hosting-Account habt (z.B. bei All-Inkl, Strato o..), ladet die Datei ins Webverzeichnis hoch.

Hier eine kurze Anleitung fr GitHub Pages:

  1. Erstellt euch bei github.com kostenlos ein Konto, falls noch nicht vorhanden.

  2. Erstellt ein neues Repository (z.B. "kanban-board").

  3. Ladet die Datei kanban.html in den Hauptordner hoch.

  4. Geht in die "Settings" des Repositories > "Pages".

  5. Whlt den Branch "main" und den Root-Ordner als Quelle aus.

  6. GitHub baut die Seite und gibt euch eine URL, z.B. https://dein-username.github.io/kanban-board/

  7. Diese URL knnt ihr gemeinsam nutzen, um jederzeit online auf das Kanban-Board zuzugreifen.
